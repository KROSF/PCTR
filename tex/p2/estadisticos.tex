\begin{equation}
    \overline{x} = \dfrac{\sum\limits_{i=1}^n{x_i}}{n}
\end{equation}
\begin{equation}
    \sigma^2 = \dfrac{\sum\limits_{i=1}^n{x^2_i}}{n}-\overline{x}^2
\end{equation}
\begin{equation}
    \sigma = \sqrt{\sigma^2}
\end{equation}
